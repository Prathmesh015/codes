    public static void addStudent(String name, String email, Integer age, Integer roll, Integer marks, String dept) {

    // READ
    public static List<Student__c> getStudents() {
        return [SELECT Id, Name, Email__c, Age__c, Roll_No__c, Marks__c, Department__c FROM Student__c];
    }
    
    
    // UPDATE
    public static void updateStudent(Id studentId, String name, String email, Integer age, Integer roll, Integer marks, String dept) {
        
        Student__c s = [SELECT Id, Name, Email__c, Age__c, Roll_No__c, Marks__c, Department__c 
                        FROM Student__c 
                        WHERE Id = :studentId LIMIT 1];
        
        s.Name = name;
        s.Email__c = email;
        s.Age__c = age;
        s.Roll_No__c = roll;
        s.Marks__c = marks;
        s.Department__c = dept;
        
        update s;
    }
    
    
    // DELETE
    public static void deleteStudent(Id studentId) {
        
        Student__c s = [SELECT Id FROM Student__c WHERE Id = :studentId LIMIT 1];
        delete s;
    }
}
